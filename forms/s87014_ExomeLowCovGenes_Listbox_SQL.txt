SELECT DISTINCT GenesHGNC_current.ApprovedSymbol As Gene, CInt(NGSWESCoverageByGene.median_20x) As [Median % coverage >20x]  FROM ((NGSTestPanelSelection INNER JOIN NGSPanelGenes ON cInt(NGSTestPanelSelection.SelectionID) = NGSPanelGenes.NGSPanelID) INNER JOIN GenesHGNC_current ON NGSPanelGenes.HGNCID = GenesHGNC_current.HGNCID) INNER JOIN NGSWESCoverageByGene ON GenesHGNC_current.HGNCID = NGSWESCoverageByGene.HGNCID WHERE NGSTestPanelSelection.SelectionType = 1 AND NGSTestPanelSelection.NGSTestID = Forms!s87014_ExomeLowCovGenes!NGSTestID AND NGSWESCoverageByGene.median_20x < 92  UNION SELECT DISTINCT GenesHGNC_current.ApprovedSymbol As Gene , CInt(NGSWESCoverageByGene.median_20x) As [Median % coverage >20x] FROM (NGSTestPanelSelection INNER JOIN GenesHGNC_current ON NGSTestPanelSelection.SelectionID = GenesHGNC_current.HGNCID) INNER JOIN NGSWESCoverageByGene ON GenesHGNC_current.HGNCID = NGSWESCoverageByGene.HGNCID WHERE NGSTestPanelSelection.SelectionType = 2 AND NGSTestPanelSelection.NGSTestID = Forms!s87014_ExomeLowCovGenes!NGSTestID AND NGSWESCoverageByGene.median_20x < 92;
